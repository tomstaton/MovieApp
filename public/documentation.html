Return a list of ALL movies to the user
app.get('/movies', (req, res) => { res.send('Successful get requests returning data on all movies'); });

Return data about a single movie by title to the user
app.get('/movies/:movieTitle', (req, res) => { res.send('Successful get request of specific movie') )};

Return data about genre
app.get('/movies/genre/:genreTitle', (req, res) => { res.send('Successful get request of specific genre') });

Return data about director
app.get('/movies/directors/:directorName', (req, res) => { res.send('Successful get request of director's info') });

Register as new user
app.post('/users/:adduser', (req, res) => { res.send('Successful post request of new user'); });

Update user name
app.put('/users/:userName', (req, res) => { res.send('Successful update request of updated username'); });

Allow user to add a movie to list of favorites
app.put('/movies/:addmovie', (req, res) => { res.send('Successful post request of new movie to list of favorites'); });

Allow user to remove a movie from list of favorites
app.delete('/movies/:movieTitle', (req, res) => { res.send('Successful delete request of movie from list of favorites'); });

Allow existing users to unregister
app.delete('/users/:userName', (req, res) => res.send('Successful delete request of user') });
